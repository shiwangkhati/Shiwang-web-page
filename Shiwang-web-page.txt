    questions.forEach((q, index) => {
      const div = document.createElement("div");
      div.className = "question";
      div.innerHTML = `<strong>${index + 1}. ${q.question}</strong><br>` +
        q.options.map(opt => `<label><input type="radio" name="q${index}" value="${opt}"> ${opt}</label><br>`).join("");
      quizDiv.appendChild(div);
    });

    function submitQuiz() {
      let score = 0;
      questions.forEach((q, index) => {
        const selected = document.querySelector(`input[name="q${index}"]:checked`);
        if (selected && selected.value === q.answer) {
          score++;
        }
      });
      document.getElementById("result").innerText = `You scored ${score} out of ${questions.length}`;
    }
